def matchingStrings(strings, queries):
    # Write your code here
    frequency= {}
    for s in strings:
        frequency[s]= frequency.get(s, 0)+ 1
    result= [frequency.get(q, 0) for q in queries]
    return result

if __name__ == '__main__':
